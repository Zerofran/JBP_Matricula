func _on_line_edit_text_changed(new_text: String) -> void:
	var lista = ObjectHUD.get_children()
	for i in lista.size():
		if (str(lista[i].get_name()).to_lower()).similarity(new_text.to_lower()) > 0.3 :
			lista[i].visible = true
		else:
			lista[i].visible = false
	if new_text == "":
		for i in lista.size():
			lista[i].visible = true
